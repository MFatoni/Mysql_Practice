DELIMITER //
CREATE PROCEDURE penilaian(
IN NIMINPUT INT(50))
BEGIN
DECLARE NILAI FLOAT;
SELECT SEMESTER_1 INTO NILAI FROM NILAI_MAHASISWA WHERE NIM = NIMINPUT;
IF NILAI < 1 THEN 
SELECT 'SERIUS KULIAH GA SIH?';
ELSEIF NILAI < 2.0 THEN 
SELECT 'KEBANYAKAN MAIN';
ELSEIF NILAI < 3.0 THEN 
SELECT 'BERUSAHA LAGI';
ELSEIF NILAI < 4 THEN 
SELECT 'MANTAP, PERTAHANKAN!';
ELSE 
SELECT 'NILAI TIDAK SESUAI';
END IF;
END//

DELIMITER ;



CALL penilaian(17080305);
CALL penilaian(17140143);
